- [ ] **Étape 3 : Créer GalerieController**

Créer `src/Controller/GalerieController.php` :

```php
<?php

namespace App\Controller;

use App\Repository\GalerieRepository;
use Symfony\Bundle\FrameworkBundle\Controller\AbstractController;
use Symfony\Component\HttpFoundation\Response;
use Symfony\Component\Routing\Attribute\Route;

class GalerieController extends AbstractController
{
    /**
     * Liste toutes les galeries disponibles.
     */
    #[Route('/galeries', name: 'app_galerie_index', methods: ['GET'])]
    public function index(GalerieRepository $galerieRepository): Response
    {
        return $this->render('galerie/index.html.twig', [
            'galeries' => $galerieRepository->findAll(),
        ]);
    }

    /**
     * Affiche une galerie avec toutes ses images.
     * Retourne 404 si la galerie n'existe pas.
     */
    #[Route('/galeries/{id}', name: 'app_galerie_show', methods: ['GET'])]
    public function show(int $id, GalerieRepository $galerieRepository): Response
    {
        $galerie = $galerieRepository->find($id);

        if (!$galerie) {
            throw $this->createNotFoundException("Galerie introuvable : $id");
        }

        return $this->render('galerie/show.html.twig', [
            'galerie' => $galerie,
        ]);
    }
}
```
